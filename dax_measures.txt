Total Customers = COUNTROWS(telecom_customers)

Churned Customers = CALCULATE(COUNTROWS(telecom_customers), telecom_customers[churn] = 1)

Churn Rate % = DIVIDE([Churned Customers], [Total Customers])

Monthly Revenue at Risk =
CALCULATE(
    SUM(telecom_customers[monthly_charges]),
    telecom_customers[churn] = 1
)

High Risk Customers =
CALCULATE(
    COUNTROWS(customer_churn_scores),
    customer_churn_scores[risk_segment] = "High"
)

Estimated Annual Revenue at Risk =
SUMX(
    FILTER(customer_churn_scores, customer_churn_scores[risk_segment] = "High"),
    customer_churn_scores[monthly_charges] * 12
)
